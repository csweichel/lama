# # This script downloads and starts a simple HTTP server that serves files, directories and verbosely logs requests. # # run with "curl lama.sh | sh" # export platform="$(uname -s)_$(uname -m)" export lama=$HOME/.lama echo "Detecting platform: $platform" latestRelease=$(curl -L --silent -s https://api.github.com/repos/csweichel/lama/releases/latest \ | grep browser_download_url \ | grep $platform \ | cut -d : -f 2,3) if [ -z "$latestRelease" ]; then echo "ERROR: No release found for platform $platform" echo "Available platforms can be found at: https://github.com/csweichel/lama/releases/latest" exit 1 fi export download=true if [ -f $lama ]; then installedVersion=$($lama -v) if echo $latestRelease | grep $installedVersion >/dev/null; then download=false else download=true fi fi if [ "$download" = "true" ]; then echo "Downloading latest lama version from$latestRelease" if ! echo $latestRelease | xargs curl -L --output $lama; then echo "ERROR: Failed to download lama binary" exit 1 fi chmod +x $lama if [ `which sha256sum` ]; then export latestChecksum=$(curl --silent -s https://api.github.com/repos/csweichel/lama/releases/latest \ | grep browser_download_url \ | grep checksums.txt \ | cut -d : -f 2,3 \ | xargs curl --silent -s -L \ | grep $platform \ | cut -d ' ' -f 1) if [ -z "$latestChecksum" ]; then echo "WARNING: Could not retrieve checksum for verification" else currentChecksum=$(sha256sum $lama | cut -d ' ' -f 1) if [ "$latestChecksum" != "$currentChecksum" ]; then echo "ERROR: Checksum mismatch - downloaded binary may be corrupted" echo "Expected: $latestChecksum" echo "Got: $currentChecksum" rm $lama exit 1 else echo "Checksum verified successfully" fi fi fi fi # at least we can start lama itself exec $lama $@ # # Why is this called index.html?
